// INSERT Operation (Recursive)
public Node insert(Node root, T data) {
    if (root == null) {
        return new Node(data);
    }
    
    if (data.compareTo(root.data) < 0) {
        root.left = insert(root.left, data);
    } else {
        root.right = insert(root.right, data);
    }
    
    return root;
}
